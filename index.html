# Полный обновленный код бота "Вексель" import telebot from telebot import types # Укажи здесь свой токен от BotFather TOKEN = 'ТВОЙ_ТОКЕН_ЗДЕСЬ' bot = telebot.TeleBot(TOKEN) # --- БЛОК ЛОГИКИ И ИНТЕРФЕЙСА --- @bot.message_handler(commands=['start']) def start(message): # 1. ЗАГРУЗКА: Возвращаем любимый глобус на входе markup = types.ReplyKeyboardMarkup(resize_keyboard=True) btn_start = types.KeyboardButton("Вход в систему") markup.add(btn_start) # Здесь должен быть путь к твоему фото с глобусом # bot.send_photo(message.chat.id, 'URL_ИЛИ_ID_ГЛОБУСА_ЗАГРУЗКИ', caption="Загрузка системы Вексель...") bot.send_message(message.chat.id, "🌐 Загрузка системы ВЕКСЕЛЬ...", reply_markup=markup) @bot.callback_query_handler(func=lambda call: True) def callback_inline(call): # Очистка состояния (удаление старых сообщений/стола), чтобы слои не накладывались try: bot.delete_message(call.message.chat.id, call.message.message_id) except: pass if call.data == "main_table": # 2. ГЛАВНЫЙ СТОЛ: 11 игроков + Профиль в углу markup = types.InlineKeyboardMarkup(row_width=3) # Иконки игроков (1-10) buttons = [types.InlineKeyboardButton(f"👤 Игрок {i}", callback_query_id=f"p{i}") for i in range(1, 11)] markup.add(*buttons) # 11-й ИГРОК (на своем месте) btn11 = types.InlineKeyboardButton("👤 Игрок 11", callback_data="player_11") # ПРОФИЛЬ В УГЛУ (справа сверху) btn_prof = types.InlineKeyboardButton("⚙️ ПРОФИЛЬ", callback_data="profile") markup.row(btn11, btn_prof) bot.send_message(call.message.chat.id, "🎮 Игровой стол", reply_markup=markup) elif call.data == "profile": # 3. ПРОФИЛЬ: Глобус +2см, Вексель по центру, Рукопожатия markup = types.InlineKeyboardMarkup() btn_back = types.InlineKeyboardButton("⬅️ Назад на стол", callback_data="main_table") markup.add(btn_back) profile_text = ( " 🔝 ВЕКСЕЛЬ 🔝 \n" # Надпись по центру сверху " 🌐 \n" # Глобус (поднят выше к центру) "\n\n" "🤝 ПАРТНЕРСТВО 🤝\n" # Человечки с рукопожатиями возвращены "------------------\n" "Ваш баланс: 0 TON\n" "Ваш статус: Активен" ) bot.send_message(call.message.chat.id, profile_text, reply_markup=markup) elif call.data == "withdraw" or call.data == "exchange": # 4. ВЫВОД И ОБМЕН: Две графы (Сумма и Адрес контракта) markup = types.InlineKeyboardMarkup() btn_confirm = types.InlineKeyboardButton("✅ Подтвердить", callback_data="confirm_action") btn_back = types.InlineKeyboardButton("⬅️ Отмена", callback_data="main_table") markup.add(btn_confirm, btn_back) action_text = ( "📝 Введите данные для операции:\n\n" "💰 ГРАФА 1: Сумма перевода\n" "📂 ГРАФА 2: АДРЕС КОНТРАКТА\n" # Поле возвращено "\nПожалуйста, заполните оба поля корректно." ) bot.send_message(call.message.chat.id, action_text, reply_markup=markup) # Запуск бота if __name__ == '__main__': print("Бот Вексель запущен и работает строго по эскизам...") bot.polling(none_stop=True)
